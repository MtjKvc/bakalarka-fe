#include <stdio.h>
#include <windows.h>                          // potrebne pre pracu s WinAPI

const unsigned char RohVpravoDole = 0xd9;     // ostatne si zistite sami!

int main(int argc, char* argv[])

{
  HANDLE  hConsole;                           // ziskanie handle konzoly
          hConsole = GetStdHandle(STD_OUTPUT_HANDLE);

                                              // Suradnice: X:stlpec, Y:riadok
  COORD point = {39,12};                      //             point.X = 39; 
                                              //            point.Y = 12;
  SetConsoleCursorPosition(hConsole, point);  // nastavi sa na [39,12]
  putchar(RohVpravoDole);                     // vytlaci tam znak '_|' 

  point.X = 0; point.Y = 22;                  // ina moznost zadania suradnice
  SetConsoleCursorPosition(hConsole, point);  // nastavi sa na [0,22]

  printf("Stlac ENTER a skoncime...\n");      // tento vypis je od [0,22] dalej
  scanf("?");
  return(0);
}
